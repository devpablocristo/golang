Valid Json:

[
   {
      "bands":{
         "members":[
            {
               "person":{
                  "name":"John",
                  "lastname": "Lennon",
                  "age":30
               }
            }
         ],
         "albums":[
            {
               "album":{
                  "bandname":"The Beatles",
                  "title":"Let It Be",
                  "year":1970
               }
            }
         ]
      }
   }
]
